141  static bool isValidAXICheckerCodes(unsigned int aOverallStatus, unsigned int aSnapshot[4], unsigned int aCumulative[4]) {
142  //Validate the codes read from the device.
143  //overallstatus could be 0 or 1
144  if (aOverallStatus > 1) return false;
145  //There are 101 possible codes and hence there cannot be any bits set beyond bit-100
146  if (aSnapshot[3]>>5 != 0) return false;
147  //There should be only 1 bit set in Snapshot
148  int nonzero_snapshots = 0;
149  for (int i=0; i<4; ++i) {
150  if (aSnapshot[i]!=0) {
151  if ((aSnapshot[i] & (aSnapshot[i]-1)) != 0) return false;
152  if (nonzero_snapshots) return false;
153  ++nonzero_snapshots;
154  //Bit set in snapshot must also be set in cumulative
155  if ((aSnapshot[i] & aCumulative[i]) == 0) return false;
156  }
157  }
158  //if snapshot is all zero then overallstatus and cumulative should be zero
159  if (!nonzero_snapshots) {
160  if (aOverallStatus) return false;
161  for (int i=0; i<4; ++i) {
162  if (aCumulative[i]) return false;
163  }
164  }
165  return true;
166  }
