42 void load_xclbin(string id, xclDeviceHandle device_handle) {
43  string xclbin_repo = search_xclbin_repository();
44  fs::path xclbin_repo_root(xclbin_repo);
45  fs::path xclbin_id(id);
46  fs::path xclbin_full_path = xclbin_repo_root / xclbin_id;
47  check_if_file_exist(xclbin_full_path.string());
48  ifstream ifs(xclbin_full_path.string());
49  ifs.seekg(0, ifs.end);
50  int size = ifs.tellg();
51  ifs.seekg(0, ifs.beg);
52  char *xclbin = new char[size];
53  ifs.read(xclbin, size);
54  check_xclbin_file_header(xclbin);
55  const xclBin *bitstream = (const xclBin *)xclbin;
56  int err = 0;
57  err = xclLockDevice(device_handle);
58  handle_errno("Lock device failed", err);
59  err = xclLoadXclBin(device_handle, bitstream);
60  handle_errno("Bitstream download failed", err);
61  err = xclUnlockDevice(device_handle);
62  handle_errno("Unlock device failed", err);
63  delete[] xclbin;
64  LOG(INFO) << "Finished downloading bitstream " << xclbin_full_path.string();
65  return;
66 }
